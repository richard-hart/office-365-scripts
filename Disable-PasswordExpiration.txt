

Connect-MgGraph -Scopes "User.ReadWrite.All","Group.ReadWrite.All"

Get-MGuser -UserId john@abc.com -Property UserPrincipalName, PasswordPolicies | Select-Object UserPrincipalName,@{N="PasswordNeverExpires";E={$_.PasswordPolicies -match "DisablePasswordExpiration"}}

Update-MgUser -UserId john@abc.com -PasswordPolicies DisablePasswordExpiration
